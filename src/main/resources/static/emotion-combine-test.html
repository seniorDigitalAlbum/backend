Facial Emotion JSON:

{
  "finalEmotion": "joy",
  "totalCaptures": 10,

  "averageConfidence": 0.85,

    {"emotion": "joy", "confidence": 0.9},
    {"emotion": "joy", "confidence": 0.8},
    {"emotion": "neutral", "confidence": 0.7}

    "슬픔": 0.05,
